on_actions = {
	on_daily_PRC = {
		effect = {
			PRC = {
				if = {
					limit = {
						PRC = {
							has_country_flag = PRC_wall_street_death
							check_variable = {
								var = PRC_readiness_level
								value < 100
							}
						}
					}
					PRC = {
						add_to_variable = {
							var = PRC_readiness_level
							value = -0.05
						}
						clamp_variable = {
							var = PRC_readiness_level
							min = 0
							max = 100
						}
					}
				}
			}
		}
	}
}
